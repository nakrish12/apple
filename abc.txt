import java.util.Scanner;

public class AbsoluteDifferenceOfDiagonals {

	public static void main(String[] args) {
		
		System.out.print("Enter number of rows");
		Scanner in = new Scanner(System.in);
		int size = in.nextInt();
		
		int data[][] = new int[size][size];
		
		for(int i = 0 ; i < size ; i++){
			
			for(int j = 0 ; j < size ; j++){
				
				System.out.print("Enter i :" + i + " j :" + j);
				data[i][j] = in.nextInt();
				
			}
			
		}
		
		System.out.print(getAbsoluteDifferenceOfDiagonals(data , size));
				
	}
	
	static int getAbsoluteDifferenceOfDiagonals(int data[][] , int size){
		
		
		int i = 0 , j = 0 , k = size - 1;
		
		int firstDiagonalSum = 0;
		int secondDiagonalSum = 0;
		
		while(i < size) {
			
			firstDiagonalSum = firstDiagonalSum + data[i][j];
			secondDiagonalSum = secondDiagonalSum + data[i][k];
			
			++i;
			++j;
			
			--k;
			
		}
		
		return Math.abs(firstDiagonalSum - secondDiagonalSum);
				
	}
	
	
	
	

}
